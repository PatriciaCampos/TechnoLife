CREATE Procedure SP_INSERTAR_CATEGORIA
(
 @id_cate INTEGER,
 @nombre_cate varchar(40),
 @activo INTEGER
)
as
begin
 insert into categoria ([id_cate], [nombre_cate], [activo] ) values (@id_cate, @nombre_cate, @activo)
end
GO



CREATE procedure SP_ACTUALIZAR_CATEGORIA
(
  @id_cate INTEGER,
  @nombre_cate varchar(40),
  @activo INTEGER
) 
as
update categoria set nombre_cate = @nombre_cate, activo = @activo
where id_cate = @id_cate
GO



CREATE procedure PROC_ELIMINAR_CATEGORIA
(
 @id_cate INTEGER 
)
as
delete from categoria 
 where @id_cate = id_cate
GO

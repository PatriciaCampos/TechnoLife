CREATE Procedure PROC_INSERTAR_SUCURSAL
(
 @id_sucursal INTEGER,
 @nombre_sucursal varchar(30)
)
as
begin
 insert into sucursal ([id_sucursal], [nombre_sucursal]) values (@id_sucursal, @nombre_sucursal)
end
GO


CREATE procedure PROC_ACTUALIZAR_SUCURSAL
(
 @id_sucursal INTEGER,
 @nombre_sucursal varchar(30)
) 
as
update sucursal set nombre_sucursal= @nombre_sucursal
where id_sucursal = @id_sucursal
GO


CREATE procedure PROC_ELIMINAR_SUCURSAL
(
@id_sucursal INTEGER
)
as
delete from sucursal 
 where @id_sucursal = id_sucursal

GO

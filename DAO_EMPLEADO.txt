--Agregar

create Procedure DA0_EMPLEADO_AGREGAR(
	@idempleado int,
	@rut varchar(30),
	@nombre varchar(60),
	@apellido varchar(30),
	@nacimiento date,
	@telefono int,
	@idcomuna int,
	@correo varchar(50),
	@activo int
)
AS
BEGIN
	set @idempleado=(Select COUNT(*)+1 from empleado)
	
	insert into empleado(id_empleado,rut_empleado,nombres,apellidos
			,fecha_nacimiento,telefono,id_comuna,correo,activo)
	values(@idempleado,@rut,@nombre,@apellido,@nacimiento,
	@telefono,@idcomuna,@correo,@activo)
END
GO




--Actualizar

create Procedure DA0_EMPLEADO_ACTUALIZAR(
	@rut int,
	@nombre varchar(60),
	@apellido varchar(30),
	@nacimiento date,
	@telef int,
	@comuna int,
	@correo varchar(50),
	@activo int
)
as
begin
update empleado set id_empleado=id_empleado, @rut=rut_empleado, @nombre=nombres,
					@apellido=apellidos,@nacimiento=fecha_nacimiento,@telef=telefono,
					@comuna=id_comuna,@correo=correo,activo=0
 where @rut=rut_empleado
end
GO




--Eliminar

create Procedure DA0_EMPLEADO_ELIMINAR(
	@emp int
)
as
begin
update empleado set id_empleado=id_empleado, rut_empleado=rut_empleado, nombres=nombres,
					apellidos=apellidos,fecha_nacimiento=fecha_nacimiento,
					id_comuna=id_comuna,correo=correo,activo=0
 where @emp=rut_empleado
end
GO
